(define-fun POST   ((b Int) (w Int)) Bool (and (= b 1) (= w 0)))

(define-fun TRANS 
            ((b Int) (w Int) (bp Int) (wp Int)) Bool
             (and (<= 0 bp) (<= 0 wp)
              (or 
                 (and (= bp ( + b 1))(= wp (- w  2)))
                 (= bp (- b 1))
                 (= bp (- b 1))
              )
             )
            )
